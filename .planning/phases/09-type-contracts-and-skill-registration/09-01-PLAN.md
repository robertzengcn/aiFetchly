```typescript
export interface GoogleMapsSearchInput {
  query: string;
  location: string;
  max_results?: number;
  include_website?: boolean;
  include_reviews?: boolean;
  show_browser?: boolean;
  proxy_ids?: number[];
}

export interface GoogleMapsBusinessResult {
  name: string;
  rating?: string;
  review_count?: number;
  category?: string;
  address?: string;
  phone?: string;
  website?: string;
  maps_url?: string;
  place_id?: string;
  hours?: string;
  latitude?: number;
  longitude?: number;
}

export interface GoogleMapsSearchResult {
  success: boolean;
  query: string;
  location: string;
  totalResults: number;
  summary: string;
  results: GoogleMapsBusinessResult[];
}

export type GoogleMapsProgressStatus =
  | "idle" | "validating" | "launching" | "navigating"
  | "loading" | "extracting" | "completed"
  | "cancelled" | "failed" | "timed_out";

export interface GoogleMapsProgressEvent {
  requestId: string;
  status: GoogleMapsProgressStatus;
  current: number;
  total: number;
  message: string;
}

export type GoogleMapsErrorCode =
  | "INVALID_INPUT" | "TIMEOUT" | "CANCELLED"
  | "SCRAPE_FAILED" | "NO_RESULTS" | "UNKNOWN";

export interface GoogleMapsErrorResponse {
  code: GoogleMapsErrorCode;
  message: string;
}

export const GOOGLE_MAPS_DEFAULT_MAX_RESULTS = 20;
export const GOOGLE_MAPS_HARD_CAP = 50;
```

    **YandexMapsSearchInput (TYPE-01):**
    - query: string -- Business keyword or category to search for (e.g. "dentist", "Italian restaurant")
    - location: string -- Target location for the search (e.g. "Moscow", "Saint Petersburg, Russia")
    - max_results?: number -- Maximum number of business results to return. Defaults to 20
    - include_website?: boolean -- Whether to extract website URLs from business listings. Defaults to true
    - include_reviews?: boolean -- Whether to include review count in results. Defaults to false
    - language?: string -- Language for Yandex Maps UI (e.g. "ru", "en", "tr"). Defaults to "ru"
    - region?: string -- Region code for search context (e.g. "ru", "kz", "by")
    - show_browser?: boolean -- Whether to show the browser window during scraping for debugging. Defaults to false
    Note: No proxy_ids field (unlike Google Maps).

    **YandexMapsBusinessResult (TYPE-02):**
    - name: string -- Business name
    - rating?: string -- Average rating as a string (e.g. "4.5")
    - review_count?: number -- Total number of reviews
    - category?: string -- Business category or type (e.g. "Dentist", "Italian Restaurant")
    - address?: string -- Physical address
    - phone?: string -- Phone number
    - website?: string -- Business website URL
    - maps_url?: string -- Yandex Maps URL for this business
    - yandex_id?: string -- Yandex business listing ID (replaces Google's place_id)
    - hours?: string -- Operating hours description
    - latitude?: number -- Geographic latitude
    - longitude?: number -- Geographic longitude

    **YandexMapsSearchResult (TYPE-03):**
    - success: boolean
    - query: string
    - location: string
    - totalResults: number
    - summary: string -- Human-readable summary of results for LLM consumption
    - results: YandexMapsBusinessResult[]

    **YandexMapsProgressStatus (TYPE-04):**
    Union type: "idle" | "validating" | "launching" | "loading" | "extracting" | "completed" | "cancelled" | "failed" | "captcha" | "timed_out"
    Note: Adds "captcha" compared to Google Maps. Removes "navigating".

    **YandexMapsProgressEvent:**

    - current: number -- Number of businesses extracted so far
    - total: number -- Target number of businesses (max_results)
